## 数据库
#### 1. MySQL 数据库
##### 1.1 数据库
- 数据库的操作
    ```shell
    # 数据库操作
    # 查看所有数据库
    show databases;
    # 进入数据库
    use databases;
    # 创建数据库
    CREATE DATABASE IF NOT EXISTS RUNOOB DEFAULT CHARSET utf8 COLLATE utf8_general_ci; 
    # 删除数据库
    drop database test01;
    # 导出数据库的数据
    # 备份test数据库
    mysqldump -u root -p 123456 test > test.sql
    # 备份所有数据库
    mysqldump -u root -p --all-databases > test.sql
    # 导入数据库的数据
    mysql -uroot -p123456 < test.sql

    # 数据表操作
    # 查看当前数据库所有表
    show tables
    # 创建数据表
    create table if not exists 表名(
    字段名称 数据类型  可选的约束条件,
    column1 datatype contrai,
    ...);
    # 修改表-添加字段
    alter table 表名 add 列名 类型 约束;
    # 例：
    alter table students add birthday datetime;

    # 修改表-修改字段类型
    alter table 表名 modify 列名 类型 约束;
    # 例：
    alter table students modify birthday date not null;

    # 修改表-修改字段名和字段类型
    alter table 表名 change 原名 新名 类型及约束;
    # 例：
    alter table students change birthday birth datetime not null;

    # 修改表-删除字段
    alter table 表名 drop 列名;
    # 例：
    alter table students drop birthday;

    # 查看表结构
    show columns from `my_datas`
    # 查看创表SQL语句
    show create table 表名;
    # 删除表
    drop table 表名;
    # 例：
    drop table students;

    # 数据表的增删改查
    # 查询数据: distinct-去重, as-取别名
    select distinct id,name from students as 表1 inner|left|right join 表2 on 表1.字段1 = 表2.字段2 
    where 条件 GROUP BY 列名 [HAVING 条件表达式] [WITH ROLLUP] 
    order by 列1 asc|desc [,列2 asc|desc,...] limit start,count;
    # 子查询
    select * from students where (age, height) =  (select max(age), max(height) from students);
    # where查询条件
    # 比较运算符: =, >, >=, <, < =, != 或 <>
    # 逻辑运算符: and, or, not
    # 模糊查询: 
    # like: 模糊查询关键字
    # %: 表示任意多个任意字符
    # _: 表示一个任意字符
    # 范围查询
    # between .. and ..: 表示在一个连续的范围内查询
    # in: 表示在一个非连续的范围内查询
    # 分页查询
    # 查询学生表，获取第n页数据的SQL语句:
    select * from students limit (n-1)*m, m;
    # 聚合函数
    # count(col): 表示求指定列的总行数
    # max(col): 表示求指定列的最大值
    # min(col): 表示求指定列的最小值
    # sum(col): 表示求指定列的和
    # avg(col): 表示求指定列的平均值
    # 分组查询
    # 统计不同性别的人的个数
    select gender,count(*) from students group by gender;

    # 插入数据
    # 主键列是自动增长，但是在全列插入时需要占位，通常使用空值(0或者null或者default)
    # 在全列插入时，如果字段列有默认值可以使用 default(类似关键字) 来占位，插入后的数据就是之前设置的默认值
    insert into students(name, height) values('刘备', 1.75),('曹操', 1.6);
    # 修改数据
    update students set age = 18, gender = '女' where id = 6;
    # 删除数据
    delete from students where id=5;
    ```

- 存储过程
```shell
# 创建存储过程
CREATE PROCEDURE 过程名 ([过程参数[,…]])
    [特性] #可选项,用于设置存储过程的行为
    存储过程体  

# 示例如下:
drop procedure if exists idata;   # 删除存储结构
delimiter ;;                      # 使用DELIMITER关键字将分隔符设置为";;"
create procedure idata()          # 创建存储结构
begin                             # 过程体开始
  declare i int;                  # 定义变量: DECLARE 变量名[,变量名...] 数据类型 [DEFAULT 值];
  set i=1;                        # 变量赋值: SET 变量名 = 表达式值[,变量名=表达式...];
START TRANSACTION;
  while(i<=1000000)do
    INSERT INTO `blog_user`(`id`, `username`, `age`, `price`, `description`, `group_id`) VALUES ( NULL,'user1111', i , "0.0",'weweweewew','1');
    set i=i+1;
  end while;
commit;
end;;                             # 过程体结束
delimiter ;                       # 还原为默认分隔符";"
call idata();                     # 调用存储过程

# 查看存储过程
show procedure status;

# 流程控制
# 条件语句: 
# IF-ELSEIF-ELSE条件语句
IF 条件 THEN 语句列表
   [ELSEIF 条件 THEN 语句列表]
   [ELSE 语句列表]
END IF;
# CASE条件语句
CASE
   WHEN 条件 THEN 语句列表
   [WHEN 条件 THEN 语句列表]
   [ELSE 语句列表]
END CASE;

# 循环语句
# WHILE循环语句
[label:] WHILE 条件 DO
   语句列表
END WHILE [label]
```
